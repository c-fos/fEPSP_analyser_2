362  coeffsLen = len(coeffs)
363  for i in range(len(coeffs)):

372  else:
373  minSD = ar.stdFinder(cD[self.deltaLen:], self.defaultFrame)
374  maxSD = ar.getLocalPtp(cD[self.deltaLen:], self.defaultFrame*0.8)
375  snr = maxSD / minSD
376  smoothCoef = minSD*(coeffTreshold*(snr**(i**(0.7)/(i**(1.5)+i+1)))+i*2)
377  logger.warn("filtering # minSD: {0}, maxSD: {1}, snr: {2}, level: {3}, smoothCoef: {4}".format(minSD, maxSD, snr, i, smoothCoef))
378  cD = pywt.thresholding.soft(cD, smoothCoef)
379  self.mysql_writer.dbWaveLevel_write(coeffTreshold, i, minSD, maxSD, smoothCoef)

595  """
