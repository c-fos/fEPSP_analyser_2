14 def iswt(coefficients, wavelet):
15  """
16  Input parameters:
17  coefficients
18  approx and detail coefficients, arranged in level value
19  exactly as output from swt:
20  e.g. [(cA1, cD1), (cA2, cD2), ..., (cAn, cDn)]
21  wavelet
22  Either the name of a wavelet or a Wavelet object
23  """
24  output = coefficients[0][0].copy() # Avoid modification of input data
25  #num_levels, equivalent to the decomposition level, n
26  num_levels = len(coefficients)
27  for j in range(num_levels,0,-1):
28  step_size = int(math.pow(2, j-1))
29  last_index = step_size
30  _, cD = coefficients[num_levels - j]
31  for first in range(last_index): # 0 to last_index - 1
32  # Getting the indices that we will transform
33  indices = arange(first, len(cD), step_size)
34  # select the even indices
35  even_indices = indices[0::2]
36  # select the odd indices
37  odd_indices = indices[1::2]
38  # perform the inverse dwt on the selected indices,
39  # making sure to use periodic boundary conditions
40  x1 = pywt.idwt(output[even_indices], cD[even_indices],wavelet, 'per')
41  x2 = pywt.idwt(output[odd_indices], cD[odd_indices],wavelet, 'per')
42  # perform a circular shift right
43  x2 = roll(x2, 1)
44  # average and insert into the correct indices
45  output[indices] = (x1 + x2)/2.
46  return output

48 def extrema(x, _max = True, _min = True, strict = False, withend = False):
49  """
50  This function will index the extrema of a given array x.
51 
52  Options:
53  _max If true, will index maxima
54  _min If true, will index minima
55  strict If true, will not index changes to zero gradient
56  withend If true, always include x[0] and x[-1]
57  This function will return a tuple of extrema indexies and values
58  """
59 
60  # This is the gradient
61 
62  dx = zeros(len(x))
63 
64  dx[1:] = diff(x)
65  dx[0] = dx[1]
66 
67  # Clean up the gradient in order to pick out any change of sign
68 
69  dx = sign(dx)
70 
71  # define the threshold for whether to pick out changes to zero gradient
72  threshold = 0
73  if strict:
74  threshold = 1
75 
76  # Second order diff to pick out the spikes
77  d2x = diff(dx)
78 
79  if _max and _min:
80  d2x = abs(d2x)
81  elif _max:
82  d2x = -d2x
83  # Take care of the two ends
84  if withend:
85  d2x[0] = 2
86  d2x[-1] = 2
87 
88  # Sift out the list of extremas
89 
90  ind = nonzero(d2x > threshold)[0]
91 
92  return ind, x[ind]

94 def smooth(x,window_len=11,window='hanning'):
95  """
96  x: the input signal
97  window_len: the dimension of the smoothing window; should be an odd integer
98  window: the type of window from 'flat', 'hanning', 'hamming', 'bartlett', 'blackman'
99  flat window will produce a moving average smoothing.
100  output:
101  the smoothed signal
102  """
103 
104  if x.ndim != 1:
105  raise ValueError, "smooth only accepts 1 dimension arrays."
106 
107  if x.size < window_len:
108  raise ValueError, "Input vector needs to be bigger than window size."
109 
110  if window_len<3:
111  return x
112 
113  if not window in ['flat','hanning', 'hamming', 'bartlett', 'blackman']:
114  raise ValueError, "Window is on of 'flat', 'hanning', 'hamming', 'bartlett', 'blackman'"
115 
116  s=r_[x[window_len-1:0:-1],x,x[-1:-window_len:-1]]
117  if window == 'flat': #moving average
118  w=ones(window_len,'d')
119  else:
120  w=eval(window+'(window_len)')
121 
122  y=convolve(w/w.sum(),s,mode='valid')
123  return y[3:]
