48 def freqLoad(filename, defFrequ, varName = "CodingFreq"):
49  try:
50  dictionary = {}
51  iniName = "{0}.ins".format(filename.strip('.dat'))
52  logger.warn("freqLoad # .ins file name: {0}".format(iniName))
53  fd = file(iniName,'r')
54  for line in fd.readlines():
55  pairs = line.split("=")
56  for variable, value in zip(pairs[::2], pairs[1::2]):
57  dictionary[variable] = value.strip()
58  frequency = int(dictionary[varName].strip()) * 1000
59  except:
60  logger.error("freqLoad # Error: {0}".format(sys.exc_info()))
61  frequency = int(defFrequ)
62  logger.warn("freqLoad # Frequency = {0}Hz".format(frequency))
63  return frequency
