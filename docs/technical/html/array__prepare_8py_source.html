20 ## приводит массив к размеру 2^x
21 #
22 # Поскольку фильтрация идет с помощью многоуровневого вейвлет разложения, то
23 # размер массива должен быть на каждом этапе разложения кратен двойке. Т.е.
24 # как минимум длинна массива должна быть кратна 2*КОЛИЧЕСТВО_УРОВНЕЙ. Но
25 # поскольку количество уровней заранее не известно, то в данной функции длинна
26 # массива наращивается до ближайшего 2^x.
27 #
28 # Дополненный участок заполняется наиболее часто встречающимся в массиве
29 # числом.
30 #
31 def dataFitting(data):
32  try:
33  dataLen = len(data)
34  tmp = 2**(math.ceil(math.log(dataLen, 2)))
35  delta = tmp - dataLen
36  meanTmp = ar.histMean(data[:dataLen / 4])
37  newData = empty(tmp, dtype='float32')
38  newData.fill(meanTmp)
39  newData[delta:] = data
40  return newData, delta
41  except:
42  logger.error("dataFitting # Error: {0}".format(sys.exc_info()))
