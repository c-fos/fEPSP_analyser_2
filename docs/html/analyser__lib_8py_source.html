347  coeffsLen = len(coeffs)
348  self.coefsBeforeF = asmatrix([coeffs[i][1] for i in range(coeffsLen)])

351  if i >= (coeffsLen - self.highNoiseLevel):
352  cD = zeros(len(cA), dtype='float32')

354  print("filtering # noisLevel: {0}".format(i))

356  minSD = ar.stdFinder(cD[self.deltaLen:], self.defaultFrame, self.argDict["debug"])
357  maxSD = ar.getLocalPtp(cD[self.deltaLen:], self.defaultFrame*0.8, self.argDict["debug"])

359  smoothCoef = minSD*(coeffTreshold*(snr**(i**(0.7)/(i**(1.5)+i+1)))+i*2)

576  def responsLength(self):
577  """
578  The function for calculation the response length - the distance between the stimuli and the moment of potential returning to baseline
579  """
580  responsMatrix=zeros((max(self.clusters),2),dtype=int)#[[start1,stop1],[start2,stop2]]
581  length=len(self.result)
582  smallFrame=self.defaultFrame/10
583  if self.argDict["debug"]==1:
584  print(len(unique(self.clusters)),len(self.clusters),"len(unique(self.clusters)),len(self.clusters)")

586  #firstSpike=self.spikeDict.values()[list(self.clusters).index(i)]
587  try:
588  lastSpike=self.spikeDict.values()[list(self.clusters).index(i+1)-1]
589  except:
590  lastSpike=self.spikeDict.values()[-1]

592  baseLevel=self.result[start-smallFrame*2:start].mean()
593  tmpObject=getattr(self,lastSpike)
594  lastMax=tmpObject.spikeMax2
595  k=lastMax
596  std2=self.result[start-smallFrame*2:start].std()
597  try:
598  while((abs(self.result[k:k+smallFrame*4].mean()-baseLevel)>std2/4 or self.result[k:k+smallFrame*4].std()>std2/4) and (k<length-smallFrame*4 and k<self.stimuli[0][i])):
599  k+=smallFrame
600  except:
601  if self.argDict["debug"]==1:
602  print "finding end of last response:", sys.exc_info()
603  k=lastMax
604  if k>length-smallFrame*4:
605  k=length-1
606  else:
607  while(abs(self.result[k:k+smallFrame*4].mean()-baseLevel)>std2/6 or self.result[k:k+smallFrame*4].std()>std2/6):
608  k+=smallFrame
609  if k>length-smallFrame*5:
610  k=length-1
611  break
612  stop=k
613  """
614  Shift the response end to left for long tail preventing
615  """
616  difference=self.result[lastMax]-self.result[stop]
617  threshold=self.result[lastMax]-0.9*difference
618  if difference>0:
619  while self.result[stop]<threshold:
620  stop-=1
621  if difference<0:
622  while self.result[stop]>threshold:
623  stop-=1
624  if self.argDict["debug"]==1:
625  print((lastMax,stop,length,"lastMax,stop,length"))

627  """
628  Calculate response end precisely using polynomial smoothing
629  """

631  if self.argDict["debug"]==1:
632  print(("respons end sample length:",sampleLen))
633  if sampleLen>0:
634  ar=polyfit(array(range(sampleLen)),self.result[lastMax:stop],2)

636  extrem=where(diff(sample)==0)[0]

638  print(("number of extremums in respons length curve:", len(extrem)))
639  if len(extrem)>0:
640  lastExtremum=extrem[-1]
641  if sample[lastExtremum]>sample[lastExtremum+1]:
642  realStop=stop
643  else:
644  realStop=lastExtremum
645  else:
646  if sample[0]>sample[1]:
647  shift=where(diff(sample)>-0.000001)[0]
648  if len(shift)>0:
649  realStop=lastMax+shift[0]
650  else:
651  realStop=stop
652  else:
653  realStop=stop
654  responsMatrix[i-1]=start,realStop
655  else:
656  responsMatrix[i-1]=start,stop
657  except:
658  responsMatrix[i-1]=start,stop

753  def epspAnaliser(self,y):
754  """
755  Function for different fEPSP properties calculation.
756  fEPSP front: "ADC codes"/"ADC points" at the 20%-80% of fEPSP amplitude interval
757  fEPSP back: "ADC codes"/"ADC points" at the 80%-20% of fEPSP amplitude interval
758  """
759  #preparing
760  maxvalue=max(y)
761  maxPoint=where(y==maxvalue)[0]
762  minvalue1=min(y[:maxPoint])
763  minvalue2=min(y[maxPoint:])
764  ampl1=maxvalue-minvalue1
765  ampl2=maxvalue-minvalue2
766  #computation
767  firstPoint=where(y[:maxPoint]>minvalue1+ampl1*0.2)[0][0]
768  firstValue=y[firstPoint]
769  secondPoint=where(y[:maxPoint]>minvalue1+ampl1*0.8)[0][0]
770  secondValue=y[secondPoint]
771  thirdPoint=maxPoint+where(y[maxPoint:]<minvalue2+ampl2*0.8)[0][0]
772  thirdValue=y[thirdPoint]
773  fourthPoint=maxPoint+where(y[maxPoint:]<minvalue2+ampl2*0.2)[0][0]
774  fourthValue=y[fourthPoint]
775  front=(secondValue-firstValue)/(secondPoint-firstPoint)
776  back=(fourthValue-thirdValue)/(fourthPoint-thirdPoint)
777  return float32(front), float32(back[0])
