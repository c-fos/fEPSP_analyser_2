45  tmp = empty(len(sample) + frame * 2)
46  result = empty(len(sample))
47  tmp[frame:-frame] = sample
48  tmp[:frame] = sample[:frame]
49  tmp[-frame:] = sample[-frame:]
50  for i in range(len(sample)):
